#include <stdio.h>
#include <stdlib.h>

int lonelyinteger(int a_count, int* a) {
    int result = 0;
    for (int i = 0; i < a_count; i++) {
        result ^= a[i];
    }
    return result;
}

int main() {
    int n;
    scanf("%d", &n);

    int *a = malloc(n * sizeof(int));
    for (int i = 0; i < n; i++) {
        scanf("%d", &a[i]);
    }

    int result = lonelyinteger(n, a);
    printf("%d\n", result);

    free(a);
    return 0;
}
