#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);
    
    int a[n];
    int count[100] = {0}; 
    
    for (int i = 0; i < n; i++) {
        scanf("%d", &a[i]);
        count[a[i]]++;
    }
    
    for (int i = 0; i < 100; i++) {
        printf("%d", count[i]);
        if (i < 99) {
            printf(" ");
        }
    }
    
    return 0;
}
