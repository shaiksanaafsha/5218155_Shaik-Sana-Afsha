#include <stdio.h>
#include <stdlib.h>

int main() {
    int n, q;
    scanf("%d %d", &n, &q);

    // Create n empty sequences
    int** arr = malloc(n * sizeof(int*));
    int* sizes = calloc(n, sizeof(int));
    int* capacities = calloc(n, sizeof(int));

    int lastAnswer = 0;

    for (int i = 0; i < n; i++) {
        arr[i] = NULL;
    }

    for (int i = 0; i < q; i++) {
        int type, x, y;
        scanf("%d %d %d", &type, &x, &y);

        int idx = (x ^ lastAnswer) % n;

        if (type == 1) {
            // Append y to arr[idx]
            if (sizes[idx] == capacities[idx]) {
                capacities[idx] = capacities[idx] == 0 ? 1 : capacities[idx] * 2;
                arr[idx] = realloc(arr[idx], capacities[idx] * sizeof(int));
            }
            arr[idx][sizes[idx]++] = y;
        } else if (type == 2) {
            // Query
            int pos = y % sizes[idx];
            lastAnswer = arr[idx][pos];
            printf("%d\n", lastAnswer);
        }
    }

    // Free memory (not mandatory for HackerRank, but good practice)
    for (int i = 0; i < n; i++) {
        free(arr[i]);
    }
    free(arr);
    free(sizes);
    free(capacities);

    return 0;
}
