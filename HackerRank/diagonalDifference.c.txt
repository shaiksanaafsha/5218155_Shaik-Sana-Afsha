#include <stdio.h>
#include <stdlib.h>

int diagonalDifference(int arr_rows, int arr_columns, int** arr) {
    int primary_sum = 0;
    int secondary_sum = 0;

    for (int i = 0; i < arr_rows; i++) {
        primary_sum += arr[i][i];                     // Primary diagonal
        secondary_sum += arr[i][arr_columns - 1 - i]; // Secondary diagonal
    }

    int diff = primary_sum - secondary_sum;
    if (diff < 0) diff = -diff; // Absolute value
    return diff;
}

int main() {
    int n;
    scanf("%d", &n);

    int **arr = malloc(n * sizeof(int *));
    for (int i = 0; i < n; i++) {
        arr[i] = malloc(n * sizeof(int));
        for (int j = 0; j < n; j++) {
            scanf("%d", &arr[i][j]);
        }
    }

    int result = diagonalDifference(n, n, arr);
    printf("%d\n", result);

    for (int i = 0; i < n; i++) {
        free(arr[i]);
    }
    free(arr);

    return 0;
}
